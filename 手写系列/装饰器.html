    # 执行时间

    function time(_target, _name, descriptor) {
    const fn = descriptor.value
    descriptor.value = function() {
        console.time()
        fn.call(this, ...arguments)
        console.timeEnd()
    }
}

class Test {
    constructor(name){
        this.showName(name)
    }
    
    @time
    showName(name) {
        console.log(`我是${name}`)
    }
}

new Test('descriptor')

    # 带参数的装饰器

    function delay(time) {
  return function (_target, _name, descriptor) {
    const fn = descriptor.value

    descriptor.value = function () {
      setTimeout(() => {
        fn.call(this)
      }, time)
    }
  }
}

class Test {
  constructor(name) {
    this.name = name
    this.showName()
  }

  @delay(1000)
  showName() {
    console.log(`我是${this.name}`)
  }
}

new Test('无来')
